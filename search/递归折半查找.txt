#include"stdio.h"
#define max 20
int data[max]={12,16,19,22,25,
32,39,48,55,57,
58,63,68,69,70,
78,84,88,90,97};
int count=0;
//递归折半查找
int b_search(int left,int right,int key)
{
	int middle;
	count++;
	if(left>right)return 0;
	else 
	{
		middle=(left+right)/2;
		if(key<data[middle])
			return b_search(left,middle-1,key);
		else if(key>data[middle])
			return b_search(middle+1,right,key);
		else if(key==data[middle])
		{printf("data[%d]=%d\n",middle,key);return 1;}
	}
	return 0;
}
//主程序
int main()
{
	int key;
	printf("Please input the key value:");
	scanf("%d",&key);
	if(b_search(0,max-1,key))
		printf("search time=%d\n",count);
	else printf("no found!!\n");
	return 0;
}
