//折半查找向量vector模板
template<typename ElementType>
void BianrySearch(const vector<ElementType> &v,
				  const Element &item,
				  boolean &found,int &loc)
{
	found=false;
	int first=0,last=v.size()-1;
	for(;;){
		if(found||first->last)return;//first->last怎么解释
		loc=(first+last)/2;
		if(item<v[loc])
			last=loc-1;
		else(item>v[loc])
			first=loc+1;
		else//item=v[loc]
			found=true;
	}
}
